    function solution(n) {
        return parseInt(n.toString(3).split('').reverse().join(''),3);
    }

    // toString을 이용해 n을 3진법으로 변환합니다. spilt('') 또는 spread syntax를 이용해 배열로 만들어줍니다. 그 다음 reverse()를 이용해 각 자릿수를 뒤집어 주고, join('')으로 문자열로 만듭니다. parseInt 메서드를 사용해 3진법인 수를 10진법으로 다시 변환해줍니다.
    

   

    function solution(n) {
        // 문자풀이
        // return (n+"").split("").reverse().map(v => parseInt(v));
    
        // 숫자풀이
        var arr = [];
    
        do {
            arr.push(n%10);
            n = Math.floor(n/10);
        } while (n>0);
    
        return arr;
    }

    // 자연수를 10으로 나눈 나머지를 빈 배열에 push하고, 자연수는 10으로 나눈 것을 Math.floor로 내림을 해주어 자연수의 각 자릿수를 원소로 가지는 배열로 만들 수 있습니다.
