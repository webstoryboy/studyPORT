    function solution(x, n) {
        var answer = [];
        for (let i = 1; i <= n; i++) {
            answer.push(x*i)
        }
        return answer;
    }

    // i를 선언 후 1을 할당하여 n개 이하 만큼 반복, i는 1씩 증가하는 반복문을 만들고 push() 함수를 사용합니다.

   
    
        function solution(x, n) {
            return Array(n).fill(x).map((v, i) => (i + 1) * v)
        }
    
        // 우선 Array()생성자로 길이가 n개인 배열을 만들고 array 메소드인 fill()을 사용하여 x로 가득채웁니다. 그리고 해당 배열을 map() 메소드를 활용하여 특정 함수 연산을 거칩니다.

x만큼 간격이 있는 n개의 숫자

함수 solution은 정수 x와 자연수 n을 입력 받아, x부터 시작해 x씩 증가하는 숫자를 n개 지니는 리스트를 리턴해야 합니다. 다음 제한 조건을 보고, 조건을 만족하는 함수, solution을 완성해주세요.


    function solution(arr) {
        return arr.reduce((sum, current) => sum + current) / arr.length;
    }

제한 조건

x는 -10000000 이상, 10000000 이하인 정수입니다.
n은 1000 이하인 자연수입니다.

입출력 예


      x	        n           answer
      2	        5           [2,4,6,8,10]
      4           3           [4,8,12]
     -4	        2           [-4,-8]
