    function solution(arr) {
        var answer = 0;
        let sum = 0;
        
        for (let i = 0; i < arr.length; i++){
         sum += arr[i];  
        }
        return answer = sum / arr.length; .
    }

    // for문을 돌려 배열의 각 인덱스에 해당하는 원소들을 모두 합해 sum에 할당하고 배열의 길이로 나눠줘 평균을 구합니다.


   
    
        function solution(arr) {
            return arr.reduce((sum, current) => sum + current) / arr.length;
          }
    
        // reduce() 메소드를 사용한 풀이 방법입니다. reduce()는 배열의 각 요소에 대해 주어진 리듀서(reducer) 함수를 실행하고, 하나의 결과값을 반환합니다.

평균 구하기

정수를 담고 있는 배열 arr의 평균값을 return하는 함수, solution을 완성해보세요.


    function solution(arr) {
        return arr.reduce((sum, current) => sum + current) / arr.length;
    }

제안 사항

arr은 길이 1 이상, 100 이하인 배열입니다.
arr의 원소는 -10,000 이상 10,000 이하인 정수입니다.

입출력 예


      arr	        return
    [1,2,3,4]	        2.5
    [5,5]	             5
