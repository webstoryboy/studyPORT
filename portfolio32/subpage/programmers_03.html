    function solution(num) {
        var answer = '';
        if (num % 2 === 0) {
            return answer = "Even"
        } else {
            return answer = "Odd"
        }
        return answer;
    }

    // 나머지 연산자인 "%"로 정수를 2로 나눈 나머지가 0일 경우 짝수이고, 0이 아닐 경우 홀수입니다.
    

   

    function solution(num) {
        return (num % 2 === 0? "Even":"Odd")
    }

    // 삼항연산자를 이용해서도 풀이할 수 있습니다. num을 2로 나눈 나머지가 0이면(true) Even을 반환하고, 그렇지 않으면(false) Odd를 반환합니다.

짝수와 홀수

정수 num이 짝수일 경우 "Even"을 반환하고 홀수인 경우 "Odd"를 반환하는 함수, solution을 완성해주세요.


    function solution(num) {
        var answer = '';
        return answer;
    }

제한 조건

num은 int 범위의 정수입니다.
0은 짝수입니다.
