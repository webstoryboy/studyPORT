let slideWrap = $(".slide-wrap");
let slideImg = $(".slide-img");
let slides = $(".slides");

let currentIndex = 0; //현재 첫 번째 보이는 이미지
let slideCount = slides.length; //이미지 갯수

//첫번째 이미지 선택해서 사라지게    slides.eq(4).hide();
//두번째 이미지 선택해서 사라지게    slides.eq(3).hide();
//세번째 이미지 선택해서 사라지게    slides.eq(2).hide();
//네번째 이미지 선택해서 사라지게    slides.eq(1).hide();
//다섯번째 이미지 선택해서 사라지게  slides.eq(0).hide();

function showNext() {

    let nextIndex = (currentIndex + 1) % slideCount;
    slides.eq(currentIndex).fadeOut(500);
    slides.eq(nextIndex).fadeIn(500);

    currentIndex = nextIndex

}
setInterval(showNext, 3000)
